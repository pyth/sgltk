464 void Mesh::draw(GLenum mode, unsigned int index_buffer,
465  glm::mat4 *model_matrix) {
466 
467  if(!shader) {
468  App::error_string.push_back("Error: No shader specified");
469  return;
470  }
471 
472  int loc;
473  glm::mat4 M;
474  glm::mat4 MV;
475  glm::mat4 MVP;
476  glm::mat4 VP;
477  glm::mat3 NM;
478  if(model_matrix)
479  M = *model_matrix;
480  else
481  M = this->model_matrix;
482 
483  NM = glm::mat3(glm::transpose(glm::inverse(M)));
484  MV = (*view_matrix) * M;
485  MVP = (*projection_matrix) * MV;
486  VP = (*projection_matrix) * (*view_matrix);
487 
488  shader->bind();
489  loc = glGetUniformLocation(shader->program,
490  model_matrix_name.c_str());
491  glUniformMatrix4fv(loc, 1, false, glm::value_ptr(M));
492 
493  loc = glGetUniformLocation(shader->program,
494  view_matrix_name.c_str());
495  glUniformMatrix4fv(loc, 1, false, glm::value_ptr(*view_matrix));
496 
497  loc = glGetUniformLocation(shader->program,
498  projection_matrix_name.c_str());
499  glUniformMatrix4fv(loc, 1, false, glm::value_ptr(*projection_matrix));
500 
501  loc = glGetUniformLocation(shader->program,
502  model_view_matrix_name.c_str());
503  glUniformMatrix4fv(loc, 1, false, glm::value_ptr(MV));
504 
505  loc = glGetUniformLocation(shader->program,
506  view_proj_matrix_name.c_str());
507  glUniformMatrix4fv(loc, 1, false, glm::value_ptr(VP));
508 
509  loc = glGetUniformLocation(shader->program,
510  normal_matrix_name.c_str());
511  glUniformMatrix3fv(loc, 1, false, glm::value_ptr(NM));
512 
513  loc = glGetUniformLocation(shader->program,
514  model_view_projection_matrix_name.c_str());
515  glUniformMatrix4fv(loc, 1, false, glm::value_ptr(MVP));
516 
517  material_uniform();
518 
519  glBindVertexArray(vao);
520  glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, ibo[index_buffer]);
521  glDrawElements(mode, num_indices[index_buffer],
