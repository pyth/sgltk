430 void Mesh::draw(GLenum mode, unsigned int index_buffer,
431  const glm::mat4 *model_matrix) {
432 
433  if(!shader) {
434  App::error_string.push_back("Error: No shader specified");
435  return;
436  }
437  shader->bind();
438 
439  glm::mat4 M;
440  glm::mat4 MV;
441  glm::mat4 MVP;
442  glm::mat4 VP;
443  glm::mat3 NM;
444  if(model_matrix)
445  M = *model_matrix;
446  else
447  M = this->model_matrix;
448  shader->set_uniform(model_matrix_name, false, M);
449 
450  NM = glm::transpose(glm::inverse(glm::mat3(M)));
451  shader->set_uniform(normal_matrix_name, false, NM);
452 
453  if(view_matrix) {
454  MV = (*view_matrix) * M;
455  shader->set_uniform(view_matrix_name, false, *view_matrix);
456  shader->set_uniform(model_view_matrix_name, false, MV);
457  }
458  if(projection_matrix) {
459  MVP = (*projection_matrix) * MV;
460  shader->set_uniform(projection_matrix_name, false,
461  *projection_matrix);
462  }
463  if(view_matrix && projection_matrix) {
464  VP = (*projection_matrix) * (*view_matrix);
465  shader->set_uniform(view_proj_matrix_name, false, VP);
466  shader->set_uniform(model_view_projection_matrix_name,
467  false, MVP);
468  }
469 
470  material_uniform();
471 
472  glBindVertexArray(vao);
473  glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, ibo[index_buffer]);
474  glDrawElements(mode, num_indices[index_buffer],
475  GL_UNSIGNED_SHORT, (void*)0);
476  glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, 0);
477  glBindVertexArray(0);
478  shader->unbind();
479 }
