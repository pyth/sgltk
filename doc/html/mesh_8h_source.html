864  pos = (glm::vec3 *)(&(vertexdata)[i] + pointer);
865  if(pos->x < bounding_box[0].x)
866  bounding_box[0].x = pos->x;
867  if(pos->y < bounding_box[0].y)
868  bounding_box[0].y = pos->y;
869  if(pos->z < bounding_box[0].z)
870  bounding_box[0].z = pos->z;

873  if(pos->y > bounding_box[1].y)
