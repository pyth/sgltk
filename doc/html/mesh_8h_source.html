433 void Mesh<Vertex>::draw(GLenum mode, unsigned int index_buffer,
434  glm::mat4 *model_matrix) {
435  if(!shader) {
436  std::cerr << "Error: No shader specified" << std::endl;
437  return;
438  }
439 
440  glm::mat4 M;
441  glm::mat4 MV;
442  glm::mat3 NM;
443  if(model_matrix)
444  M = *model_matrix;
445  else
446  M = this->model_matrix;
447 
448  MV = (*view_matrix) * M;
449  NM = glm::mat3(glm::transpose(glm::inverse(MV)));
450  glm::mat4 MVP = (*projection_matrix) * MV;
451 
452  shader->bind();

454  model_view_matrix_name);
455  glUniformMatrix4fv(loc, 1, false, glm::value_ptr(MV));
456

458  normal_matrix_name);

460
