795 int Mesh::attach_vertex_buffer(const std::vector<T>& vertexdata,
796  GLenum usage) {
797  GLuint buf;
798  glGenBuffers(1, &buf);
799  vbo.push_back(buf);
800 
801  unsigned int buffer_size = vertexdata.size() * sizeof(T);
802 
803  glBindBuffer(GL_ARRAY_BUFFER, buf);
804  glBufferData(GL_ARRAY_BUFFER, buffer_size, vertexdata.data(), usage);
805 
806  vertex_buffer_size_map[vbo.size() - 1] = buffer_size;
807  vertex_buffer_usage_map[vbo.size() - 1] = usage;
808  return vbo.size() - 1;

820 template <typename T>
821 bool Mesh::replace_buffer_data(unsigned int buffer_index,
822  const std::vector<T>& data) {
823  if(buffer_index >= vbo.size()) {
824  App::error_string.push_back("The value of the variable"
825  "buffer_index if greater than the number"
826  "of vertex buffers.");
827  return false;
828  }
829 
830  unsigned int current_size = vertex_buffer_size_map[buffer_index];
831  unsigned int new_size = data.size() * sizeof(T);
832 
833  glBindBuffer(GL_ARRAY_BUFFER, vbo[buffer_index]);
834  if(current_size == new_size) {
835  //replace the buffer data without reallocation
836  glBufferSubData(GL_ARRAY_BUFFER, 0, new_size, data.data());
837  } else {
838  //replace the buffer data with reallocation
839  glBufferData(GL_ARRAY_BUFFER, new_size, data.data(),
840  vertex_buffer_usage_map[buffer_index]);
841  }
842 
843  return true;
844 }
