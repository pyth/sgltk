714 int Mesh::attach_vertex_buffer(const std::vector<T> *vertexdata,
715  GLenum usage) {
716  GLuint buf;
717  glGenBuffers(1, &buf);
718  vbo.push_back(buf);
719 
720  unsigned int buffer_size = vertexdata->size() * sizeof(T);
721 
722  glBindBuffer(GL_ARRAY_BUFFER, buf);
723  glBufferData(GL_ARRAY_BUFFER, buffer_size, vertexdata->data(), usage);
724 
725  vertex_buffer_size_map[vbo.size() - 1] = buffer_size;
726  vertex_buffer_usage_map[vbo.size() - 1] = usage;
727  return vbo.size() - 1;

739 template <typename T>
740 bool Mesh::replace_buffer_data(unsigned int buffer_index, std::vector<T> *data) {
741  if(buffer_index >= vbo.size()) {
742  App::error_string.push_back("The value of the variable"
743  "buffer_index if greater than the number"
744  "of vertex buffers.");
745  return false;
746  }
747 
748  unsigned int current_size = vertex_buffer_size_map[buffer_index];
749  unsigned int new_size = data->size() * sizeof(T);
750 
751  glBindBuffer(GL_ARRAY_BUFFER, vbo[buffer_index]);
752  if(current_size == new_size) {
753  //replace the buffer data without reallocation
754  glBufferSubData(GL_ARRAY_BUFFER, 0, new_size, data->data());
755  } else {
756  //replace the buffer data with reallocation
757  glBufferData(GL_ARRAY_BUFFER, new_size, data->data(),
758  vertex_buffer_usage_map[buffer_index]);
759  }
760 
761  return true;
762 }
