867 template <typename T>
868 bool Mesh::replace_buffer_data(unsigned int buffer_index,
869  const std::vector<T>& data) {
870 
871  if(buffer_index >= vbo.size()) {
872  App::error_string.push_back("The value of the variable"
873  "buffer_index if greater than the number"
874  "of vertex buffers.");
875  return false;
876  }
877 
878  glBindBuffer(GL_ARRAY_BUFFER, vbo[buffer_index]);
879 
880  int usage;
881  int current_size = vbo_sizes[buffer_index];
882  int new_size = data.size() * sizeof(T);
883  glGetBufferParameteriv(GL_ARRAY_BUFFER, GL_BUFFER_USAGE, &usage);
884 
885  if(current_size == new_size) {
886  //replace the buffer data without reallocation
887  glBufferSubData(GL_ARRAY_BUFFER, 0, new_size, data.data());
888  } else {
889  //replace the buffer data with reallocation
890  glBufferData(GL_ARRAY_BUFFER, new_size, data.data(), usage);
891  vbo_sizes[buffer_index] = new_size;
892  }
893 
894  return true;
895 }
