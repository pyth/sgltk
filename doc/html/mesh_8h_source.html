772 int Mesh::attach_vertex_buffer(const std::vector<T> *vertexdata,
773  GLenum usage) {
774  GLuint buf;
775  glGenBuffers(1, &buf);
776  vbo.push_back(buf);
777 
778  unsigned int buffer_size = vertexdata->size() * sizeof(T);
779 
780  glBindBuffer(GL_ARRAY_BUFFER, buf);
781  glBufferData(GL_ARRAY_BUFFER, buffer_size, vertexdata->data(), usage);
782 
783  vertex_buffer_size_map[vbo.size() - 1] = buffer_size;
784  vertex_buffer_usage_map[vbo.size() - 1] = usage;
785  return vbo.size() - 1;

797 template <typename T>
798 bool Mesh::replace_buffer_data(unsigned int buffer_index, std::vector<T> *data) {
799  if(buffer_index >= vbo.size()) {
800  App::error_string.push_back("The value of the variable"
801  "buffer_index if greater than the number"
802  "of vertex buffers.");
803  return false;
804  }
805 
806  unsigned int current_size = vertex_buffer_size_map[buffer_index];
807  unsigned int new_size = data->size() * sizeof(T);
808 
809  glBindBuffer(GL_ARRAY_BUFFER, vbo[buffer_index]);
810  if(current_size == new_size) {
811  //replace the buffer data without reallocation
812  glBufferSubData(GL_ARRAY_BUFFER, 0, new_size, data->data());
813  } else {
814  //replace the buffer data with reallocation
815  glBufferData(GL_ARRAY_BUFFER, new_size, data->data(),
816  vertex_buffer_usage_map[buffer_index]);
817  }
818 
819  return true;
820 }
