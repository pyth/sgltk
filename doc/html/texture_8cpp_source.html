146 void Texture::load_texture(const Image& image) {
147  if(!image.image)
148  return;
149 
150  SDL_Surface *tmp = SDL_ConvertSurfaceFormat(image.image,
151  SDL_PIXELFORMAT_RGBA8888, 0);
152 
153  glActiveTexture(GL_TEXTURE0);
154  glBindTexture(target, texture);
155  glTexParameteri(target, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
156  glTexParameteri(target, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
157  glTexParameteri(target, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
158  glTexParameteri(target, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
159  glTexImage2D(target, 0, GL_RGBA, image.width, image.height, 0,
160  GL_RGBA, GL_UNSIGNED_INT_8_8_8_8, tmp->pixels);
161  glGenerateMipmap(target);
162  glBindTexture(target, 0);
163  SDL_FreeSurface(tmp);
164 }
