73 void App::poll_events() {

75  SDL_Event event;
76 
77  while(SDL_PollEvent(&event)) {
78  switch(event.type) {
79  case SDL_QUIT:
80  running = false;
81  break;
82  case SDL_WINDOWEVENT:
83  if(event.window.event == SDL_WINDOWEVENT_RESIZED) {
84  width = event.window.data1;
85  height = event.window.data2;
86  handle_resize();
87  }
88  break;
89  case SDL_KEYDOWN:
90  case SDL_KEYUP:
91  break;
92  case SDL_MOUSEWHEEL:
93  handle_mouse_wheel(event.wheel.x, event.wheel.y);
94  break;
95  case SDL_MOUSEBUTTONDOWN:
96  case SDL_MOUSEBUTTONUP:
97  if(mouse_relative)
98  handle_mouse_button(0, 0, (MOUSE_BUTTON)event.button.button,
99  (event.button.state == SDL_PRESSED),
100  event.button.clicks);
101  else
102  handle_mouse_button(event.button.x, event.button.y,
103  (MOUSE_BUTTON)event.button.button,
104  (event.button.state == SDL_PRESSED),
105  event.button.clicks);
106  break;
107  case SDL_MOUSEMOTION:
108  if(mouse_relative)
109  handle_mouse_motion(event.motion.xrel,
110  event.motion.yrel);
