66 void Window::poll_events() {
67  SDL_Event event;
68 
69  while(SDL_PollEvent(&event)) {
70  switch(event.type) {
71  case SDL_QUIT:
72  handle_exit();
73  break;
74  case SDL_WINDOWEVENT:
75  if(event.window.event == SDL_WINDOWEVENT_CLOSE) {
76  handle_exit();
77  }
78  if(event.window.event == SDL_WINDOWEVENT_RESIZED) {
79  width = event.window.data1;
80  height = event.window.data2;
81  handle_resize();
82  }
83  break;
84  case SDL_KEYDOWN:
85  case SDL_KEYUP:
86  break;
87  case SDL_MOUSEWHEEL:
88  handle_mouse_wheel(event.wheel.x, event.wheel.y);
89  break;
90  case SDL_MOUSEBUTTONDOWN:
91  case SDL_MOUSEBUTTONUP:
92  if(mouse_relative)
93  handle_mouse_button(0, 0, (MOUSE_BUTTON)event.button.button,
94  (event.button.state == SDL_PRESSED),
95  event.button.clicks);
96  else
97  handle_mouse_button(event.button.x, event.button.y,
98  (MOUSE_BUTTON)event.button.button,
99  (event.button.state == SDL_PRESSED),
100  event.button.clicks);
101  break;
102  case SDL_MOUSEMOTION:
103  if(mouse_relative)
104  handle_mouse_motion(event.motion.xrel,
105  event.motion.yrel);
106  else
107  handle_mouse_motion(event.motion.x,
108  event.motion.y);
109  break;
110  }
111  }

153 void Window::run(int fps) {
154  double frame_time;
155  Timer frame_timer;
156  if(fps < 1)
157  frame_time = 1e-30;
158  else
159  frame_time = 1000.0 / fps;
160  bool running = true;
161 
162  while(running) {
163  frame_timer.start();

165  if(!window) {
166  break;
167  }
168  display();
169  delta_time = frame_timer.get_time();
170  if(fps > 0) {

173  std::this_thread::sleep_for(std::chrono::duration<double, std::milli>(frame_time - delta_time));
174  }
175  }
