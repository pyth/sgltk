160  switch(event.type) {
161  case SDL_QUIT:
162  handle_exit();
163  break;
164  case SDL_WINDOWEVENT:
165  switch(event.window.event) {
166  case SDL_WINDOWEVENT_CLOSE:
167  handle_exit();
168  break;
169  case SDL_WINDOWEVENT_RESIZED:
170  width = event.window.data1;
171  height = event.window.data2;
172  handle_resize();
173  break;
174  }
175  break;
176  case SDL_KEYDOWN:
177  if(event.key.repeat == 0) {
178  keys_pressed.push_back(SDL_GetKeyName(event.key.keysym.sym));
179  handle_key_press(SDL_GetKeyName(event.key.keysym.sym), true);
180  }
181  break;
182  case SDL_KEYUP:
183  keys_pressed.erase(std::find(keys_pressed.begin(),
184  keys_pressed.end(),
185  SDL_GetKeyName(event.key.keysym.sym)));
186  handle_key_press(SDL_GetKeyName(event.key.keysym.sym), false);
187  break;
188  case SDL_MOUSEWHEEL:
189  handle_mouse_wheel(event.wheel.x, event.wheel.y);
190  break;
191  case SDL_MOUSEBUTTONDOWN:
192  case SDL_MOUSEBUTTONUP:
193  if(mouse_relative)
194  handle_mouse_button(0, 0, event.button.button,
195  (event.button.state == SDL_PRESSED),
196  event.button.clicks);
197  else
198  handle_mouse_button(event.button.x, event.button.y,
199  event.button.button,
200  (event.button.state == SDL_PRESSED),
201  event.button.clicks);
202  break;
203  case SDL_MOUSEMOTION:
204  if(mouse_relative)
205  handle_mouse_motion(event.motion.xrel,
206  event.motion.yrel);
207  else
208  handle_mouse_motion(event.motion.x,
209  event.motion.y);
210  break;
211  case SDL_CONTROLLERDEVICEADDED:
212  gamepad = new Gamepad(event.cdevice.which);
213  gamepad_instance_id_map[gamepad->instance_id] = gamepad;
214  handle_gamepad_added(gamepad->id);
215  break;
216  case SDL_CONTROLLERDEVICEREMOVED:
217  gamepad_it = gamepad_instance_id_map.find(event.cdevice.which);
218  if(gamepad_it != gamepad_instance_id_map.end()) {
219  gamepad = gamepad_it->second;
220  gamepad_instance_id_map.erase(gamepad->instance_id);
221  handle_gamepad_removed(gamepad->id);
222  delete Gamepad::id_map[gamepad->id];
223  }
224  break;
225  case SDL_CONTROLLERBUTTONDOWN:
226  gamepad_it = gamepad_instance_id_map.find(event.cdevice.which);
227  if(gamepad_it != gamepad_instance_id_map.end()) {
228  gamepad = gamepad_it->second;
229  gamepad->set_button_state(event.cbutton.button, true);
230  handle_gamepad_button_press(gamepad->id,
231  event.cbutton.button, true);
232  }
233  break;
234  case SDL_CONTROLLERBUTTONUP:
235  gamepad_it = gamepad_instance_id_map.find(event.cdevice.which);
236  if(gamepad_it != gamepad_instance_id_map.end()) {
237  gamepad = gamepad_it->second;
238  gamepad->set_button_state(event.cbutton.button, false);
239  handle_gamepad_button_press(gamepad->id,
240  event.cbutton.button, false);
241  }
242  break;
243  case SDL_CONTROLLERAXISMOTION:
244  gamepad_it = gamepad_instance_id_map.find(event.cdevice.which);
245  if(gamepad_it != gamepad_instance_id_map.end()) {
246  gamepad = gamepad_it->second;
247  handle_gamepad_axis_change(gamepad->id,
248  event.caxis.axis,
249  gamepad->get_axis_value(event.caxis.axis));
250  }
251  break;
252  case SDL_JOYDEVICEADDED:
253  if(!SDL_IsGameController(event.jdevice.which)) {
254  joystick = new Joystick(event.jdevice.which);
255  joystick_instance_id_map[joystick->instance_id] = joystick;
256  handle_joystick_added(joystick->id);
257  }
258  break;
259  case SDL_JOYDEVICEREMOVED:

261  if(joystick_it != joystick_instance_id_map.end()) {
262  joystick = joystick_it->second;
263  joystick_instance_id_map.erase(joystick->instance_id);
264  handle_joystick_removed(joystick->id);
265  delete Joystick::id_map[joystick->id];
266  }
267  break;
268  case SDL_JOYBUTTONDOWN:
269  joystick_it = joystick_instance_id_map.find(event.jdevice.which);
270  if(joystick_it != joystick_instance_id_map.end()) {
271  joystick = joystick_it->second;
272  joystick->set_button_state(event.jbutton.button, true);
273  handle_joystick_button_press(joystick->id,
274  event.jbutton.button, true);
275  }
276  break;
277  case SDL_JOYBUTTONUP:
278  joystick_it = joystick_instance_id_map.find(event.jdevice.which);
279  if(joystick_it != joystick_instance_id_map.end()) {
280  joystick = joystick_it->second;
281  joystick->set_button_state(event.jbutton.button, false);
282  handle_joystick_button_press(joystick->id,
283  event.jbutton.button, false);
284  }
285  break;
286  case SDL_JOYAXISMOTION:
287  joystick_it = joystick_instance_id_map.find(event.jdevice.which);
288  if(joystick_it != joystick_instance_id_map.end()) {
289  joystick = joystick_it->second;
290  handle_joystick_axis_change(joystick->id,
291  event.jaxis.axis,
292  joystick->get_axis_value(event.jaxis.axis));
293  }
294  break;
295  case SDL_JOYHATMOTION:
296  joystick_it = joystick_instance_id_map.find(event.jdevice.which);
297  if(joystick_it != joystick_instance_id_map.end()) {
298  joystick = joystick_it->second;
299  handle_joystick_hat_change(joystick->id,
300  event.jhat.hat,
301  event.jhat.value);
302  }
303  break;
304  case SDL_JOYBALLMOTION:
305  joystick_it = joystick_instance_id_map.find(event.jdevice.which);
306  if(joystick_it != joystick_instance_id_map.end()) {
307  joystick = joystick_it->second;
308  handle_joystick_ball_motion(joystick->id,
309  event.jball.ball,
310  event.jball.xrel,
311  event.jball.yrel);
312  }
313  break;
314  }

342  }

424 void Window::run(unsigned int fps) {
425  double frame_time;
426  double time_to_wait;
427  Timer frame_timer;
428  if(fps < 1)
429  frame_time = 0;
430  else
431  frame_time = 1000.0 / fps;
432  running = true;
433 
434  display();
435 
436  while(running) {
437  poll_events();
438  if(!window) {
439  break;
440  }
441  frame_timer.start();
442  display();
443  if(fps > 0) {
444  time_to_wait = frame_time - frame_timer.get_time_ms();
445  if(delta_time < frame_time) {
446  std::this_thread::sleep_for(std::chrono::duration<double, std::milli>(time_to_wait));
447  }
448  }
449  delta_time = frame_timer.get_time_s();
450  SDL_GL_SwapWindow(window);
451  }
452 }
