104 void Window::poll_events() {
105  SDL_Event event;
106  Gamepad *gamepad;
107 
108  while(SDL_PollEvent(&event)) {
109  switch(event.type) {
110  case SDL_QUIT:
111  handle_exit();
112  break;
113  case SDL_WINDOWEVENT:
114  switch(event.window.event) {
115  case SDL_WINDOWEVENT_CLOSE:
116  handle_exit();
117  break;
118  case SDL_WINDOWEVENT_RESIZED:
119  width = event.window.data1;
120  height = event.window.data2;
121  handle_resize();
122  break;
123  }
124  break;
125  case SDL_KEYDOWN:
126  case SDL_KEYUP:
127  break;
128  case SDL_MOUSEWHEEL:
129  handle_mouse_wheel(event.wheel.x, event.wheel.y);
130  break;
131  case SDL_MOUSEBUTTONDOWN:
132  case SDL_MOUSEBUTTONUP:
133  if(mouse_relative)
134  handle_mouse_button(0, 0, (MOUSE_BUTTON)event.button.button,
135  (event.button.state == SDL_PRESSED),
136  event.button.clicks);
137  else
138  handle_mouse_button(event.button.x, event.button.y,
139  (MOUSE_BUTTON)event.button.button,
140  (event.button.state == SDL_PRESSED),
141  event.button.clicks);
142  break;
143  case SDL_MOUSEMOTION:
144  if(mouse_relative)
145  handle_mouse_motion(event.motion.xrel,
146  event.motion.yrel);
147  else
148  handle_mouse_motion(event.motion.x,
149  event.motion.y);
150  break;
151  case SDL_CONTROLLERDEVICEADDED:
152  gamepad = new sgltk::Gamepad(event.cdevice.which);
153  handle_gamepad_added(gamepad->id);
154  break;
155  case SDL_CONTROLLERDEVICEREMOVED:
156  gamepad = sgltk::Gamepad::instance_id_map[event.cdevice.which];
157  handle_gamepad_removed(gamepad->id);
158  delete sgltk::Gamepad::id_map[gamepad->id];
159  break;
160  case SDL_CONTROLLERBUTTONDOWN:
161  case SDL_CONTROLLERBUTTONUP:
162  gamepad = sgltk::Gamepad::instance_id_map[event.cdevice.which];
163  handle_gamepad_button(gamepad->id,
164  event.cbutton.button,
165  (event.cbutton.state == SDL_PRESSED));
166  break;
167  case SDL_CONTROLLERAXISMOTION:
168  gamepad = sgltk::Gamepad::instance_id_map[event.cdevice.which];
169  handle_gamepad_axis(gamepad->id,
170  event.caxis.axis,
171  event.caxis.value);
172  }
173  }
174  handle_keyboard();
175 }

227 void Window::run(int fps) {
