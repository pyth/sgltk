117 void Window::poll_events() {
118  int value;
119  SDL_Event event;
120  Gamepad *gamepad;
121  std::vector<int>::iterator button;
122 
123  while(SDL_PollEvent(&event)) {
124  switch(event.type) {
125  case SDL_QUIT:
126  handle_exit();
127  break;
128  case SDL_WINDOWEVENT:
129  switch(event.window.event) {
130  case SDL_WINDOWEVENT_CLOSE:
131  handle_exit();
132  break;
133  case SDL_WINDOWEVENT_RESIZED:
134  width = event.window.data1;
135  height = event.window.data2;
136  handle_resize();
137  break;
138  }
139  break;
140  case SDL_KEYDOWN:
141  if(event.key.repeat == 0) {
142  keys_pressed.push_back(SDL_GetKeyName(event.key.keysym.sym));
143  handle_key_press(SDL_GetKeyName(event.key.keysym.sym), true);
144  }
145  break;
146  case SDL_KEYUP:
147  keys_pressed.erase(std::find(keys_pressed.begin(),
148  keys_pressed.end(),
149  SDL_GetKeyName(event.key.keysym.sym)));
150  handle_key_press(SDL_GetKeyName(event.key.keysym.sym), false);
151  break;
152  case SDL_MOUSEWHEEL:
153  handle_mouse_wheel(event.wheel.x, event.wheel.y);
154  break;
155  case SDL_MOUSEBUTTONDOWN:
156  case SDL_MOUSEBUTTONUP:
157  if(mouse_relative)
158  handle_mouse_button(0, 0, event.button.button,
159  (event.button.state == SDL_PRESSED),
160  event.button.clicks);
161  else
162  handle_mouse_button(event.button.x, event.button.y,
163  event.button.button,
164  (event.button.state == SDL_PRESSED),
165  event.button.clicks);
166  break;
167  case SDL_MOUSEMOTION:
168  if(mouse_relative)
169  handle_mouse_motion(event.motion.xrel,
170  event.motion.yrel);
171  else
172  handle_mouse_motion(event.motion.x,
173  event.motion.y);
174  break;
175  case SDL_CONTROLLERDEVICEADDED:
176  gamepad = new sgltk::Gamepad(event.cdevice.which);
177  handle_gamepad_added(gamepad->id);
178  break;
179  case SDL_CONTROLLERDEVICEREMOVED:
180  gamepad = sgltk::Gamepad::instance_id_map[event.cdevice.which];
181  handle_gamepad_removed(gamepad->id);
182  delete sgltk::Gamepad::id_map[gamepad->id];
183  break;
184  case SDL_CONTROLLERBUTTONDOWN:
185  gamepad = sgltk::Gamepad::instance_id_map[event.cdevice.which];
186  gamepad->buttons_pressed.push_back(event.cbutton.button);
187  handle_gamepad_button_press(gamepad->id,
188  event.cbutton.button,
189  (event.cbutton.state == SDL_PRESSED));
190  break;
191  case SDL_CONTROLLERBUTTONUP:
192  gamepad = sgltk::Gamepad::instance_id_map[event.cdevice.which];
193  button = std::find(gamepad->buttons_pressed.begin(),
194  gamepad->buttons_pressed.end(),
195  event.cbutton.button);
196  gamepad->buttons_pressed.erase(button);
197  handle_gamepad_button_press(gamepad->id,
198  event.cbutton.button,
199  (event.cbutton.state == SDL_PRESSED));
200  break;
201  case SDL_CONTROLLERAXISMOTION:
202  gamepad = sgltk::Gamepad::instance_id_map[event.cdevice.which];
203  handle_gamepad_axis_change(gamepad->id,
204  event.caxis.axis,
205  event.caxis.value);
206  break;
207  }
208  }
209 
210  for(auto key : keys_pressed) {
211  handle_keyboard(key);
212  }
213 
214  for(auto device : sgltk::Gamepad::instance_id_map) {
215  if(!device.second)
216  continue;
217  for(unsigned int axis = 0; axis < device.second->num_axes; axis++) {
218  value = SDL_GameControllerGetAxis(device.second->gamepad, (SDL_GameControllerAxis)axis);
219  handle_gamepad_axis(device.second->id, axis, value);
220  }
221  for(unsigned int button = 0; button < device.second->buttons_pressed.size(); button++) {
222  handle_gamepad_button(device.second->id, device.second->buttons_pressed[button]);
223  }
224  }

279 void Window::run(int fps) {
280  double frame_time;
281  Timer frame_timer;
282  if(fps < 1)
283  frame_time = 1e-30;
284  else
285  frame_time = 1000.0 / fps;
286  running = true;
287 
288  while(running) {
289  frame_timer.start();
290  poll_events();
291  if(!window) {
292  break;
293  }
294  display();
295  delta_time = frame_timer.get_time();
296  if(fps > 0) {
297  if(delta_time < frame_time) {
298  //SDL_Delay(frame_time - delta_time);
299  std::this_thread::sleep_for(std::chrono::duration<double, std::milli>(frame_time - delta_time));
300  }
301  }
302  delta_time = frame_timer.get_time();
303  SDL_GL_SwapWindow(window);
304  }
