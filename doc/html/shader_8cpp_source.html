29 bool Shader::attach_file(const std::string& filename, GLenum type) {
30  std::string path;
31  GLint compiled;
32  char infoLog[4096];
33  int infoLogLength;
34  SDL_RWops *file;
35  const char *code;
36  GLuint tmp;
37  GLint size;
38 
39  if((filename.length() > 1 && filename[0] == '/') ||
40  (filename.length() > 2 && filename[1] == ':')) {
41  //absolute path
42  file = SDL_RWFromFile(filename.c_str(), "r+b");
43  path = filename;
44  } else {
45  //relative path
46  for(unsigned int i = 0; i < paths.size(); i++) {
47  file = SDL_RWFromFile((paths[i] + filename).c_str(),
48  "r+b");
49  if(file) {
50  path = paths[i] + filename;
51  break;
52  }
53  }
54  }
55 
56  if(!file) {
57  App::error_string.push_back(std::string("Could not open shader file: ")
58  + filename + std::string(" - ") + SDL_GetError());
59  return false;
60  }
61  size = (int)SDL_RWsize(file);
62  std::vector<char> buf(size);
63  SDL_RWread(file, buf.data(), size, 1);
64  SDL_RWclose(file);
65  code = buf.data();
66 
67  tmp = glCreateShader(type);
68  glShaderSource(tmp, 1, &code, &size);
69  glCompileShader(tmp);
70  glGetShaderiv(tmp, GL_COMPILE_STATUS, &compiled);
71  if(!compiled) {
72  glGetShaderInfoLog(tmp, sizeof(infoLog), &infoLogLength,
73  infoLog);
74  if(infoLogLength > 0) {
75  printf("CompileShader() infoLog %s \n%s\n",
76  filename.c_str(), infoLog);
77  return false;
78  }
79  }
80 
81  glAttachShader(program, tmp);
82  if(modify) {
83  shader_path_map[path] = type;
84  }
85 
86  glDeleteShader(tmp);
87  return true;
88 }
